避免使用 exception 來建立運作流程的控制

  • 效率不好的寫法
try{
    DoSomething();
}catch(CustomExceptionA e){
    GoToCaseA(e);
}catch(CustomExceptionB e){
    GoToCaseB(e);
}
  • 效率好的寫法
public enum ProcessCase{
    A,
    B
};

var result = DoSomething();
switch(result){
    case ProcessCase.A:
        GoToCaseA();
        break;
    case ProcessCase.B:
        GoToCaseB();
        break;
    default:
        break;
}

比較

以下程式可以用 isUseException 來控制是否要觸發 excpetion

當使用以下程式連續觸發 1000 次 exception 時，理論上就能明顯感受到執行效率低下

以我自己的電腦來測，光是 1000 次就要跑將近5秒

class Program
{
    static void Main(string[] args)
    {
        DateTime st = DateTime.Now;
        var loop = 1000;
        //var isUseException = false;
        var isUseException = true;
        for (int i = 0; i < loop; i++)
        {
            Run(isUseException);
        }
        DateTime end = DateTime.Now;
        var runningTime = end - st;
        Console.WriteLine($"Running: {runningTime}");
    }

    private static void Run(bool isUseException)
    {
        if (isUseException)
        {
            try
            {
                RunCore();
            }
            catch (Exception)
            {
            }                
        }
    }

    private static void RunCore()
    {
        throw new AAAException();
    }
}


class AAAException : Exception
{
    public AAAException() { }
}
