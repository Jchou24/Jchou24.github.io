精簡語法

?: Ternary Conditional Operator

  • 直覺寫法
if( isSuccess ){
    someVar = 1
}else{
    someVar = 0
}
  • 一行寫法
someVar = isSuccess ? 1 : 0;

?? Null Coalescing Operator

string x = "AAA";
string y = x ?? "123"; // y => "AAA"

string x = null;
string y = x ?? "123"; // y => "123"

?. Null Conditional Operators

  • 直覺寫法
string customerName = null;
if( customer != null ){
    customerName = customer.Name;
}
  • 一行寫法
string customerName = customer?.Name;

x?.y?.z Null Propagation

  • 直覺寫法
string customerAddressCode = null;
if( customer != null ){
    if( customer.Address != null ){
        customerAddressCode = customer.Address.Code;
    }
}
  • 一行寫法
string customerAddressCode = customer?.Address?.Code;

() => {} Arrow Function

  • 直覺寫法
public class DateTimeModel
{
    private DateTime dateTime;

    public string GetISOString()
    {
        return this.dateTime.ToUniversalTime().ToString("s");
    }

    public DateTimeModel(DateTime dateTime)
    {
        this.dateTime = dateTime;
    }
}
  • 一行寫法
public class DateTimeModel
{
    private DateTime dateTime;

    public string GetISOString() => this.dateTime.ToUniversalTime().ToString("s");

    public DateTimeModel(DateTime dateTime)
    {
        this.dateTime = dateTime;
    }
}

short circuit

  • 直覺寫法
string s = null;
if( s != null ){
    if( s.ToLower() == "aaa" ){
        // do something
    }
}
  • 一行寫法
string s = null;
if( s != null && s.ToLower() == "aaa" ){
    // do something
}

降低誤用機會

constructor塞值，最好在property/field前面加上this

public class AAA{
    string bbb;
    string BBB { get; set; }
    AAA(string bbb){
        this.BBB = bbb;
        this.bbb = bbb;
    }
}
  • 邏輯錯誤的寫法
public class AAA{
    string bbb;
    string BBB { get; set; }
    AAA(string bbb){
        BBB = bbb; // BBB 有正確賦值
        bbb = bbb; // bbb "沒有"正確賦值，但compiler並不會報錯
    }
}

避免直接鏈式存取 Property

  • 危險的寫法
var x = a.b.c; // 當b是null時，會有exception
  • 安全的寫法
var x = a?.b?.c;
